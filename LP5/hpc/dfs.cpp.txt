#include <iostream>
#include <vector>
#include <omp.h>

using namespace std;

void parallel_dfs_util(vector<vector<int>> &graph, int node, vector<bool> &visited) {
    bool alreadyVisited;
    
    // Critical section to safely check and update the visited array
    #pragma omp critical
    {
        alreadyVisited = visited[node];
        if (!visited[node]) {
            visited[node] = true;
            cout << node << " ";
        }
    }
    
    if (alreadyVisited) return;
    
    // Parallelize the exploration of neighbors
    #pragma omp parallel for
    for (int i = 0; i < graph[node].size(); i++) {
        int neighbor = graph[node][i];
        if (!visited[neighbor]) {
            parallel_dfs_util(graph, neighbor, visited);
        }
    }
}

void parallel_dfs(vector<vector<int>> &graph, int start, int V) {
    vector<bool> visited(V, false);
    cout << "Parallel DFS Traversal: ";
    parallel_dfs_util(graph, start, visited);
    cout << endl;
}

int main() {
    int V, E;
    cout << "Enter number of vertices and edges: ";
    cin >> V >> E;
    
    vector<vector<int>> graph(V);
    
    cout << "Enter edges (u v):\n";
    for (int i = 0; i < E; i++) {
        int u, v;
        cin >> u >> v;
        graph[u].push_back(v);
        graph[v].push_back(u); // undirected graph
    }
    
    int start;
    cout << "Enter starting node: ";
    cin >> start;
    
    parallel_dfs(graph, start, V);
    
    return 0;
}
